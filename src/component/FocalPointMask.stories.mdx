import { useRef, useEffect } from 'react';
import { Canvas, Meta, Story, ArgsTable } from '@storybook/addon-docs/blocks';
import { argTypes } from '../../docs/argTypes';
import FocalPointMask from './FocalPointMask';

export const startWebComponents = () => {
  if (!window.customElements.get('focal-point-mask')) {
    window.customElements.define('focal-point-mask', FocalPointMask);
  }
}
startWebComponents();

export const TemplateFactory = mediaType => ({ onResize, ...props}) => {
  const ref = useRef();
  useEffect(() => {
    const element = ref.current;
    if (element != null) {
      element.onResize = evt => {
        const { width, height } = evt.detail[0].contentRect;
        onResize(`${width}Ã—${height} px`);
      };
    }
  }, []);
  return (
    <focal-point-mask ref={ref} {...props}>
      {mediaType === 'image' && (
        <img src="https://picsum.photos/id/1012/3840/2160"/>
      )}
      {mediaType === 'video' && (
        <video autoPlay muted loop>
          <source src="https://file-examples-com.github.io/uploads/2017/04/file_example_MP4_1280_10MG.mp4"/>
        </video>
      )}
    </focal-point-mask>
  );
}

<Meta title="Examples" argTypes={argTypes}/>
